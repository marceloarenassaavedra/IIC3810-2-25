\begin{enumerate}
\item Considere el siguiente lenguaje:
  \begin{multline*}
    L \ = \ \{ M \mid M \text{ es una MT determinista con alfabeto de
      entrada } \Sigma \text{ y existe un polinomio } p \\ \text{ tal
      que para todo } w \in \Sigma^*, \text{ el número de pasos de } M
    \text{ con entrada } w \text{ es menor o igual a } p(|w|)\}
  \end{multline*}
  Demuestre que $L$ es un lenguaje indecidible.

  \item Sea $\CC$ una clase de complejidad de lenguajes decidibles que
    es cerrada bajo reducciones many-to-one de tiempo polinomial.
    Demuestre que si $\CC$ tiene un problema completo bajo este tipo
    de reducciones, entonces $\CC$ tiene una sintaxis efectiva.
\end{enumerate}
